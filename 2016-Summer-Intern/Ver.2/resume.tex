\vspace{1em}
\cventry{}{\textsf{网站日志分析系统}}{}{}{}{
	通过分布式hadoop集群处理网站产生的大量日志。通过分布式存储，按小时保存并以小时为周期定时将上一小时的数据同步到日志分析机上。统计数据按小时更新，统计项包括关键词搜索量PV，网站的浏览量PV，类别访问量，访客数UV（包括新访客数、新访客比例），跳出率等信息供网站管理者分析网站使用。
	\begin{description}
		\item[\textsf{系统支持}] Ubuntu Server + JDK-1.8 + CDH-5.4.1 + Hive-0.9.0 + MySQL5
		\item[\textsf{系统实现}] \hfill 
			\begin{enumerate}
				\item 通过Shell命令把日志数据导入到HDFS中 
				\item 使用MapReduce对数据进行清洗,再存储到HDFS中
				\item 使用Hive对清洗后的数据进行多维分析
				\item 使用Sqoop把分析结果导入到MySQL中
				\item 返回链接给Web前端，将结果以图表，表格等多种形式表示出来。
			\end{enumerate} %itemize
		\item[\textsf{个人职责}] 独立完成。
	\end{description}
}
